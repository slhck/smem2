 150class MemData(Proc):

 153    def __init__(self):
 154        """Initializes MemData and parses /proc/meminfo."""
 155        self._memdata = {}
 156
 157        regex = re.compile("(?P<name>\\S+):\\s+(?P<amount>\\d+) kB")
 158        for line in self.readlines("meminfo"):
 159            match = regex.match(line)
 160            if match:
 161                self._memdata[match.group("name").lower()] = int(match.group("amount"))

 164        """Retrieves a value from the stored meminfo data.

 168
 169        Returns:
 170            int: The value from /proc/meminfo in kilobytes.
 171        """
 172        return self._memdata[entry]

 331def totalmem(config: SmemConfig):
 332    """Calculates the total memory to be used as a reference for percentages.
 333
 334    It can be based on real memory from dmidecode, a user-provided value,
 335    or the 'MemTotal' value from /proc/meminfo.
 336
 337    Args:
 338        config (SmemConfig): The smem2 configuration.
 339
 340    Returns:
 341        int: The total memory in kilobytes.
 342    """
 343    if config.realmem:
 344        return fromunits(config.realmem) / 1024
 345    else:
 346        if config.system or config.sysdetail:
 347            try:
 348                # only for systems running directly on HW
 349                ram = (
 350                    os.popen(
 351                        "dmidecode --type memory 2> /dev/null|awk '/Size: [0-9]+ MB/{sum+=$2};END{print sum}'",
 352                    )
 353                    .read()
 354                    .strip()
 355                )
 356                if ram.isdigit():
 357                    return fromunits(str(int(ram)) + "M") / 1024
 358            except Exception:
 359                pass
 360
 361        return MemData()("memtotal")

 364def pidmaps(pid, proc: ProcessData, config: SmemConfig, nomaps=False):
 365    """Parses the memory maps for a given process.
 366
 367    It reads /proc/<pid>/smaps or /proc/<pid>/smaps_rollup to get memory
 368    usage information for each mapping.
 369
 370    Args:
 371        pid (int or str): The process ID.
 372        proc (ProcessData): The ProcessData instance.
 373        config (SmemConfig): The smem2 configuration.
 374        nomaps (bool, optional): If True, only read the first mapping entry,
 375            used for feature detection. Defaults to False.
 376
 377    Returns:
 378        dict: A dictionary of memory mappings, where keys are start addresses.
 379    """
 380    maps: dict = {}
 381    start = None
 382    status = None
 383    if config.rollup or config.pssdetail or config.rssdetail:
 384        status = proc.readlines("%s/status" % pid)
 385
 386    for l in proc.mapdata(pid):
 387        f = l.split()
 388        if f[-1] == "kB":
 389            maps[start][f[0][:-1].lower()] = int(f[1])
 390        elif "-" in f[0] and ":" not in f[0]:  # looks like a mapping range
 391            start, end = f[0].split("-")
 392            start = int(start, 16)
 393            name = "<anonymous>"
 394            if len(f) > 5:
 395                name = f[5]
 396                if config.basename:
 397                    name = os.path.basename(name)
 398            maps[start] = dict(
 399                end=int(end, 16),
 400                perm=f[1],
 401                offset=int(f[2], 16),
 402                major=int(f[3].split(":")[0], 16),
 403                minor=int(f[3].split(":")[1], 16),
 404                inode=int(f[4]),
 405                name=name,
 406                pss=0,
 407                rss=0,
 408                uss=0,
 409                swap=0,
 410                swappss=0,
 411                pcent=0,
 412                size=0,
 413            )
 414            if config.pssdetail:
 415                (
 416                    maps[start]["pss_anon"],
 417                    maps[start]["pss_file"],
 418                    maps[start]["pss_shmem"],
 419                ) = (0, 0, 0)
 420
 421            # for nomaps mode: stop reading mappings
 422            if nomaps:
 423                break
 424
 425    # add pss, rss, uss if available from smaps
 426    if not config.rollup:
 427        if not config.pssdetail:
 428            for m in maps:
 429                maps[m]["uss"] = maps[m].get("private_clean", 0) + maps[m].get(
 430                    "private_dirty", 0
 431                )
 432                maps[m]["pss"] = maps[m]["uss"] + maps[m].get("shared_clean", 0) / 2
 433                maps[m]["rss"] = maps[m].get("shared_clean", 0) + maps[m].get(
 434                    "shared_dirty", 0
 435                )
 436        if config.rssdetail and status:
 437            # from /proc/pid/status
 438            for s in status:
 439                if "RssAnon" in s:
 440                    maps[start]["rss_anon"] = int(s.split()[1])
 441                elif "RssFile" in s:
 442                    maps[start]["rss_file"] = int(s.split()[1])
 443                elif "RssShmem" in s:
 444                    maps[start]["rss_shmem"] = int(s.split()[1])
 445
 446        # Parse VmSize for VSS calculation
 447        if status:
 448            for s in status:
 449                if "VmSize:" in s:
 450                    # Set size for the last mapping (rollup case) or all mappings
 451                    if config.rollup and start:
 452                        maps[start]["size"] = int(s.split()[1])
 453                    else:
 454                        # For non-rollup, distribute VmSize across all mappings
 455                        total_size = int(s.split()[1])
 456                        if maps:
 457                            size_per_map = (
 458                                total_size // len(maps) if len(maps) > 0 else 0
 459                            )
 460                            for m in maps:
 461                                maps[m]["size"] = size_per_map
 462                    break
 463
 464    # Apply mapfilter if specified
 465    if config.mapfilter and not nomaps:
 466        f = {}
 467        for m in maps:
 468            if not filters(config.mapfilter, m, config, lambda x: maps[x]["name"]):
 469                f[m] = maps[m]
 470        return f
 471    return maps

 474def maptotals(pids, proc: ProcessData, config: SmemConfig):
 475    """Aggregates memory usage data for a list of pids, including map details.
 476
 477    For each process, it gets memory totals (PSS, RSS, USS, etc.) and
 478    detailed mapping information. It applies process and user filters.
 479
 480    Args:
 481        pids (list[int]): A list of process IDs.
 482        proc (ProcessData): The ProcessData instance.
 483        config (SmemConfig): The smem2 configuration.
 484
 485    Returns:
 486        dict: A dictionary where keys are process IDs and values are dictionaries
 487              containing memory totals and mapping details for each process.
 488    """
 489    pt: dict = {}
 490    for p in pids:
 491        if filters(
 492            config.processfilter, p, config, proc.pidcmd, proc.pidtostr, proc.pidname
 493        ) or filters(config.userfilter, p, config, proc.pidusername):
 494            continue
 495        maps = pidmaps(p, proc, config)
 496        if maps:
 497            pt[p] = {}
 498            pt[p]["maps"] = maps
 499            totals = pidtotals(p, proc, config)
 500            # Flatten the totals into the main dict for compatibility with display code
 501            pt[p].update(totals)
 502    return pt

 505def pidtotals(pid, proc: ProcessData, config: SmemConfig):
 506    """Calculates the total memory usage (PSS, RSS, USS, etc.) for a single process.
 507
 508    It sums up the values from all memory mappings of the process.
 509
 510    Args:
 511        pid (int or str): The process ID.
 512        proc (ProcessData): The ProcessData instance.
 513        config (SmemConfig): The smem2 configuration.
 514
 515    Returns:
 516        dict: A dictionary with total PSS, RSS, USS, Swap, etc., for the process.
 517    """
 518    totals: dict = {}
 519    maps = pidmaps(pid, proc, config)
 520    if maps:
 521        maplist = maps.values()
 522        totals["pss"] = sum([m.get("pss", 0) for m in maplist])
 523        totals["rss"] = sum([m.get("rss", 0) for m in maplist])
 524        private_clean = sum(m.get("private_clean", 0) for m in maplist)
 525        private_dirty = sum(m.get("private_dirty", 0) for m in maplist)
 526        totals["uss"] = private_clean + private_dirty
 527        totals["swap"] = sum([m.get("swap", 0) for m in maplist])
 528        totals["size"] = sum([m.get("size", 0) for m in maplist])
 529        totals["maps"] = len(maps)
 530        if config.swappss:
 531            totals["swappss"] = sum([m.get("swappss", 0) for m in maplist])
 532        if config.pssdetail:
 533            totals["pss_anon"] = sum([m.get("pss_anon", 0) for m in maplist])
 534            totals["pss_file"] = sum([m.get("pss_file", 0) for m in maplist])
 535            totals["pss_shmem"] = sum([m.get("pss_shmem", 0) for m in maplist])
 536        if config.rssdetail:
 537            totals["rss_anon"] = sum([m.get("rss_anon", 0) for m in maplist])
 538            totals["rss_file"] = sum([m.get("rss_file", 0) for m in maplist])
 539            totals["rss_shmem"] = sum([m.get("rss_shmem", 0) for m in maplist])
 540    return totals

 543def usertotals(pids, proc: ProcessData, config: SmemConfig):
 544    """Aggregates memory usage by user.
 545
 546    It groups processes by user and calculates the total memory usage for each user.
 547
 548    Args:
 549        pids (list[int]): A list of process IDs to analyze.
 550        proc (ProcessData): The ProcessData instance.
 551        config (SmemConfig): The smem2 configuration.
 552
 553    Returns:
 554        dict: A dictionary where keys are user IDs and values are dictionaries
 555              containing process lists and memory totals for each user.
 556    """
 557    ut: dict = {}
 558    for p in pids:
 559        if filters(
 560            config.processfilter, p, config, proc.pidcmd, proc.pidtostr, proc.pidname
 561        ) or filters(config.userfilter, p, config, proc.pidusername):
 562            continue
 563        u = proc.piduser(p)
 564        if u not in ut:
 565            ut[u] = {"pids": [], "name": "?"}
 566            ut[u]["name"] = proc.username(u)
 567        ut[u]["pids"].append(p)
 568
 569    # get totals per user
 570    for u in ut:
 571        ut[u]["totals"] = processtotals(ut[u]["pids"], proc, config)
 572    return ut

 575def cmdtotals(pids, proc: ProcessData, config: SmemConfig):
 576    """Aggregates memory usage by command line.
 577
 578    It groups processes by their command line and calculates the total memory
 579    usage for each command.
 580
 581    Args:
 582        pids (list[int]): A list of process IDs to analyze.
 583        proc (ProcessData): The ProcessData instance.
 584        config (SmemConfig): The smem2 configuration.
 585
 586    Returns:
 587        dict: A dictionary where keys are command lines and values are
 588              dictionaries containing process lists and memory totals.
 589    """
 590    # totals per command
 591    ct: dict = {}
 592
 593    for p in pids:
 594        if filters(
 595            config.processfilter, p, config, proc.pidcmd, proc.pidtostr, proc.pidname
 596        ) or filters(config.userfilter, p, config, proc.pidusername):
 597            continue
 598        cmdline = proc.pidcmd(p)
 599        if not cmdline:
 600            continue
 601
 602        parts = cmdline.split()
 603        if not parts:
 604            continue
 605        c = parts[0]
 606
 607        # if multiple processes per command, then a list of pids
 608        if c in ct:
 609            ct[c]["pids"].append(p)
 610        else:
 611            # first pid for this command
 612            ct[c] = {"pids": [p]}
 613
 614    # get totals per command
 615    for c in ct:
 616        ct[c]["totals"] = processtotals(ct[c]["pids"], proc, config)
 617        if "totals" in ct[c] and ct[c]["totals"]:
 618            ct[c]["name"] = c
 619            ct[c]["user"] = proc.pidusername(ct[c]["pids"][0])
 620
 621    # remove commands with empty totals
 622    for c in list(ct.keys()):
 623        if "totals" not in ct[c] or not ct[c]["totals"]:
 624            del ct[c]
 625    return ct

 643def units(x):
 644    """Converts a value in bytes to a human-readable string with units (K, M, G).
 645
 646    Args:
 647        x (int or float): The value in bytes.
 648
 649    Returns:
 650        str: A human-readable string representation of the value.
 651    """
 652    if x > 1024 * 1024 * 1024:
 653        return "%.1fG" % (x / (1024.0 * 1024 * 1024))
 654    if x > 1024 * 1024:
 655        return "%.1fM" % (x / (1024.0 * 1024))
 656    if x > 1024:
 657        return "%.1fK" % (x / 1024.0)
 658    return str(x)

 661def fromunits(x):
 662    """Converts a human-readable string with units (K, M, G, etc.) to bytes.
 663
 664    Args:
 665        x (str): The string with a value and unit (e.g., "1024M").
 666
 667    Returns:
 668        int: The value in bytes.
 669
 670    Raises:
 671        SystemExit: If the unit is not recognized.
 672    """
 673    s = dict(
 674        k=2**10,
 675        K=2**10,
 676        kB=2**10,
 677        KB=2**10,
 678        M=2**20,
 679        MB=2**20,
 680        G=2**30,
 681        GB=2**30,
 682        T=2**40,
 683        TB=2**40,
 684    )
 685    for k, v in list(s.items()):
 686        if x.endswith(k):
 687            return int(float(x[: -len(k)]) * v)

 750def mapnametotals(pids, proc: ProcessData, config: SmemConfig):
 751    """Aggregates memory usage by mapping name across a list of processes.
 752
 753    It iterates through processes and their memory maps, summing up memory
 754    statistics for each unique mapping name.
 755
 756    Args:
 757        pids (list[int]): A list of process IDs.
 758        proc (ProcessData): The ProcessData instance.
 759        config (SmemConfig): The smem2 configuration.
 760
 761    Returns:
 762        dict: A dictionary where keys are mapping names and values are
 763              dictionaries of aggregated memory statistics.
 764    """
 765    totals: dict = {}
 766    for pid in pids:
 767        # Apply filters like the original version
 768        if filters(
 769            config.processfilter, pid, config, proc.pidcmd, proc.pidtostr, proc.pidname
 770        ) or filters(config.userfilter, pid, config, proc.pidusername):
 771            continue
 772        try:
 773            maps = pidmaps(pid, proc, config)
 774            seen = {}
 775            for m in list(maps.keys()):
 776                name = maps[m]["name"]
 777                if name not in totals:
 778                    t = dict(
 779                        size=0,
 780                        rss=0,
 781                        pss=0,
 782                        shared_clean=0,
 783                        shared_dirty=0,
 784                        private_clean=0,
 785                        count=0,
 786                        private_dirty=0,
 787                        referenced=0,
 788                        swap=0,
 789                        pids=0,
 790                    )
 791                    if config.swappss:
 792                        t["swappss"] = 0
 793                else:
 794                    t = totals[name]
 795                for k in t:
 796                    t[k] += maps[m].get(k, 0)
 797                t["count"] += 1
 798                if name not in seen:
 799                    t["pids"] += 1
 800                    seen[name] = 1
 801                totals[name] = t
 802        except (OSError, IOError):
 803            continue
 804    return totals

 839def get_system_data(proc: ProcessData, config: SmemConfig):
 840    """Collects and returns system-wide memory usage data.
 841
 842    This provides a breakdown of memory usage by different system components
 843    like kernel, firmware, userspace, etc.
 844
 845    Args:
 846        proc (ProcessData): The ProcessData instance.
 847        config (SmemConfig): The smem2 configuration.
 848
 849    Returns:
 850        list[tuple]: A list of tuples, where each tuple represents a memory
 851                     area and its usage in PSS and RSS.
 852    """
 853    t = totalmem(config)
 854    m = MemData()
 855    mt = m("memtotal")
 856    f = m("memfree")
 857    kernel = kernelsize(config)
 858
 859    # total amount mapped into userspace (ie mapped an unmapped pages) /dev/zero mappings
 860    u = m("anonpages") + m("mapped")
 861
 862    # total amount used by hardware
 863    fh = int(max(t - mt - kernel, 0))
 864
 865    # total amount allocated by kernel not for userspace
 866    kd = mt - f - u
 867
 868    # total amount in kernel caches
 869    kdc = m("buffers") + m("sreclaimable") + m("cached") - m("mapped")
 870
 871    if config.sysdetail:
 872        shmp = mapshared(proc, config)  # shared memory mapped by processes
 873        mapzero = mapdevzero(
 874            proc, config
 875        )  # dev/zero mapped as files should be subtracted from files Mapped from userspace
 876        mapped = m("mapped") - mapzero
 877        modules = kernelmodsize(proc, config)
 878        shm = m("shmem") - shmp
 879
 880        mounts = proc.readlines("mounts")
 881        ramfs = 0
 882        for r in mounts:
 883            if "ramfs" in r:
 884                try:
 885                    ramfs += int(
 886                        (os.popen("du -sk %s" % r.split()[1]).read()).split()[0]
 887                    )
 888                except Exception:
 889                    pass
 890        filecache = m("cached") - mapped - shm - ramfs
 891        unknown = (
 892            mt
 893            - f
 894            - u
 895            - (
 896                modules
 897                + m("pagetables")
 898                + m("kernelstack")
 899                + m("slab")
 900                + m("buffers")
 901                + filecache
 902                + shm
 903                + ramfs
 904            )
 905        )
 906        if unknown < 0:
 907            unknown = 0
 908
 909    lines: list[tuple]
 910    if config.sysdetail:
 911        lines = [
 912            ("firmware/hardware", fh, 0, 0),
 913            ("kernel image", kernel, 0, 0),
 914            ("kernel modules", modules, 0, 0),
 915            ("page tables", m("pagetables"), 0, 0),
 916            ("kernel stack", m("kernelstack"), 0, 0),
 917            ("slab (all/SReclaimable)", m("slab"), m("sreclaimable"), 0),
 918            ("buffers", m("buffers"), m("buffers"), 0),
 919            ("cached (w/o mapped,tmpfs,ramfs)", filecache, filecache, 0),
 920            ("shared (non process tmpfs)", shm, 0, 0),
 921            ("ramfs", ramfs, 0, 0),
 922            ("unknown", unknown, 0, 0),
 923            ("processes (all/mapped files)", u, mapped, 0),
 924            ("free memory", f, f, 0),
 925        ]
 926        if "all" in config.columns or "details" in config.columns:
 927            line_all: list[tuple] = [
 928                ("-------------------------------", 0, 0, 0),
 929                ("/dev/zero mapped", 0, 0, mapzero),
 930                ("shared by processes", 0, 0, shmp),
 931                ("unevictable", 0, 0, m("unevictable")),
 932                ("dirty (unwritten to file)", 0, 0, m("dirty")),
 933                ("swapped", 0, 0, m("swaptotal") - m("swapfree")),
 934                ("available (estimated)", 0, 0, m("memavailable")),
 935            ]
 936            lines = lines + line_all
 937    else:
 938        lines = [
 939            ("firmware/hardware", fh, 0),
 940            ("kernel image", kernel, 0),
 941            ("kernel dynamic memory", kd, kdc),
 942            ("userspace memory", u, m("mapped")),
 943            ("free memory", f, f),
 944        ]
 945    return lines
